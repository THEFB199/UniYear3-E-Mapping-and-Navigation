32/* Return a handle to device instance given device names */



33static MW_videoInfo_t *getDeviceHandle(const uint8_T *devName)



34{



35	int devIndx = -1;



36	unsigned int i;



37



38	/* We have to search all entries since blocks can be called out of order */



39	for (i = 0; i < MW_NUM_MAX_VIDEO_DEVICES; i++) {



40		if ((hVideoDev[i].fd != 0) && (strcmp((const char *)hVideoDev[i].devName, (const char *)devName) == 0)) {



41			devIndx = i;



42			break;



43		}



44	}



45



46	if (devIndx == -1) {



47		return NULL;



48	}



49	else {



50        return(&hVideoDev[i]);



51	}



52}

54/* Return number of bytes required to store a single horizontal line */



55/* of image given format */



56uint32_T getBytesPerLine(uint32_T pixelFormat, uint32_T width)



57{



58	uint32_T bytesPerLine;



59



60	switch (pixelFormat) {



61		case MW_YCBCR422:



62			bytesPerLine = 2 * width;



63			break;



64		case MW_RGB:



65			bytesPerLine = 3 * width;



66			break;



67		default:



68            bytesPerLine = 0;



69			break;



70	}



71



72	return bytesPerLine;



73}

105/* Create a new device instance */



106static MW_videoInfo_t *newDevice(void)



107{



108    MW_videoInfo_t *h;



109	uint32_T i;



110



111    /* Find an unused video device */



112	h = NULL;



113	for (i = 0; i < MW_NUM_MAX_VIDEO_DEVICES; i++) {



114		if (hVideoDev[i].fd == 0) {



115			h = &hVideoDev[i];



116			memset(h, 0, sizeof(MW_videoInfo_t));



117			h->fd = -1; /* -1 indicates device is not opened yet  */



118			break;



119		}



120	}



121	if (h == NULL) {



122		MW_ERROR_EXIT1("Too many video devices. Number of video devices must be smaller than %d\n",



123			MW_NUM_MAX_VIDEO_DEVICES);



124	}



125



126	return h;



127}

129/* Destroy a device instance */



130static void destroyDevice(MW_videoInfo_t *h)



131{



132	if (h != NULL) {



133	    memset(h, 0, sizeof(MW_videoInfo_t));



134	}



135}

137/* Check device parameters for validity */



138/* 1. Each video device must use a unique device name */



139/* 2. Live video capture is not supported in Windows  */



140static void checkDeviceParams(const uint8_T *devName, uint32_T simOutput, uint32_T pixelFormat, uint32_T imWidth)



141{



142    MW_videoInfo_t *h;



143



144#if (defined(MATLAB_MEX_FILE) || defined(RSIM_WITH_SL_SOLVER)) && !defined(__linux__)



145	/* Check if live video capture in simulation is supported */



146	if (simOutput == SIM_OUTPUT_LIVE_VIDEO) {



147	    MW_ERROR_EXIT("Live video capture is not supported on Windows platforms. "



148			"Select a simulation output that generates a synthetic image.\n");



149	}



150#endif



151



152	/* Check if we have a device instance with the same name */



153	h = getDeviceHandle(devName);



154	if (h != NULL) {



155	    MW_ERROR_EXIT1("Each Video Capture block must use a unique Video device name. "



156			"Video device name %s is already used.", devName);



157	}



158



159	/* Check that pixel format is in range */



160	if ((pixelFormat != MW_YCBCR422) && (pixelFormat != MW_RGB)) {



161		MW_ERROR_EXIT1("Unknown Pixel format %d.", pixelFormat);



162	}



163



164	/* Image width must be even in YCbCr 4:2:2 mode */



165	if ((pixelFormat == MW_YCBCR422) && ((imWidth & 0x1) != 0)) {



166		MW_ERROR_EXIT1("%s: Image width must be even when Pixel order is YCbCr 4:2:2. "



167			"This format has half as many chroma samples as that of luminance plane.",  devName);



168	}



169}

171/* Initialization routine for video capture module */



172void MW_videoCaptureInit



173(



174        const uint8_T *devName, /* p1 */



175        int32_T roiTop,         /* p2 */



176        int32_T roiLeft,        /* p3 */



177        int32_T roiWidth,       /* p4 */



178        int32_T roiHeight,      /* p5 */



179        uint32_T imWidth,       /* p6 */



180        uint32_T imHeight,      /* p7 */



181        uint32_T pixelFormat,   /* p8 */



182        uint32_T pixelOrder,    /* p9 */



183        uint32_T simOutput,     /* p10 */



184        real_T sampleTime       /* p11 */



185)



186{



187    MW_videoInfo_t *h;



188    unsigned int i;



189



190	/* Check device parameters for validity */



191	checkDeviceParams(devName, simOutput, pixelFormat, imWidth);



192



193	/* Create a new video device instance */



194    h                       = newDevice();



195    h->devName = (uint8_T*)malloc(strlen((const char*)devName)+1);



196    if (h->devName == NULL) {



197        MW_ERROR_EXIT("Memory allocation error.");



198    }



199	strcpy((char*)h->devName, (const char*)devName);



200    h->roi.top              = roiTop;         /* Stores cropping rectangle (ROI from block mask) */



201	h->roi.left             = roiLeft;



202	h->roi.width            = roiWidth;



203	h->roi.height           = roiHeight;



204    h->imFormat.width       = imWidth;        /* Capture width */



205    h->imFormat.height      = imHeight;       /* Capture height */



206	h->imFormat.pln12Width  = 0;              /* Used for chroma sub-sampled formats */



207	h->imFormat.pixelFormat = pixelFormat;    /* Stores image size and pixel format */



208    h->pixelOrder           = pixelOrder;     /* Interleave / planar selection */



209	h->hRgbRefLine          = NULL;



210    h->frameRate            = 1.0/sampleTime; /* Frame rate */



211    h->simOutput            = simOutput;      /* Generated output in simulation */



212	h->frmCount             = 0;



213    h->v4l2BufCount         = 0;



214    h->v4l2CaptureStarted   = 0;



215    for (i = 0; i < MW_NUM_V4L2_BUFFERS; i++) {



216        h->frm[i].start  = NULL;



217        h->frm[i].length = 0;



218    }



219



220#ifdef MATLAB_MEX_FILE



221	/* Register cleanup-function */



222	mexAtExit(videoCaptureCleanup);



223#endif



224    



225	/* Open and initialize device */



226#if (defined(MATLAB_MEX_FILE) || defined(RSIM_WITH_SL_SOLVER)) && defined(__linux__) 



227	/* Simulation && Linux */



228	if (h->simOutput == SIM_OUTPUT_LIVE_VIDEO) {



229#ifdef _MW_V4L2_DEBUG_LVL2_



230        printf("\nOpening V4L2 device %s\n", devName);



231#endif



232	    openV4L2Device(h);



233	    initV4L2Device(h);



234	}



235	else {



236		openSimDevice(h);



237		initSimDevice(h);



238	}



239#elif (defined(MATLAB_MEX_FILE) || defined(RSIM_WITH_SL_SOLVER)) && !defined(__linux__) 



240	/* Normal or Rapid Accelerator && not Linux */



241    openSimDevice(h);



242    initSimDevice(h);



243#else



244	/* Code generation */



245    openV4L2Device(h);



246	initV4L2Device(h);



247#endif



248}

250/* Device termination function */



251static void terminateDevice(MW_videoInfo_t *h)



252{



253	/* De-allocate buffers and close device file */



254    if (h != NULL) {



255#if (defined(MATLAB_MEX_FILE) || defined(RSIM_WITH_SL_SOLVER)) && defined(__linux__) 



256	    if (h->simOutput == SIM_OUTPUT_LIVE_VIDEO) {



257	        closeV4L2Device(h);



258	    }



259	    else {



260		    closeSimDevice(h);



261	    }



262#elif (defined(MATLAB_MEX_FILE) || defined(RSIM_WITH_SL_SOLVER)) && !defined(__linux__)



263	    closeSimDevice(h);



264#else



265	    /* We are generating code */



266	    closeV4L2Device(h);



267#endif



268        if(h->devName) {



269            free(h->devName);



270        }



271        destroyDevice(h);



272    }



273}

275/* Simulink interface for video capture terminate function */



276void MW_videoCaptureTerminate(const uint8_T *devName)



277{



278    MW_videoInfo_t *h = getDeviceHandle(devName);



279



280#ifdef _MW_V4L2_DEBUG_LVL2_



281	printf("\nTerminate for device %s\n", devName);



282#endif



283    if (h != NULL) {



284        terminateDevice(h);



285    }



286}

289/* Outputs an image frame */



290/* In simulation mode either a synthetic image or live video (on Linux only) */



291/* is output. In code generation, always outputs live video */



292void MW_videoCaptureOutput(const uint8_T *devName, uint8_T *pln0, uint8_T *pln1, uint8_T *pln2)



293{



294	MW_videoInfo_t *h = getDeviceHandle(devName);



295



296#if (defined(MATLAB_MEX_FILE) || defined(RSIM_WITH_SL_SOLVER)) && defined(__linux__) 



297	if (h->simOutput == SIM_OUTPUT_LIVE_VIDEO) {



298        readV4L2Frame(h, pln0, pln1, pln2);



299	}



300	else {



301        genSimOutput(h, pln0, pln1, pln2);



302	}



303#elif (defined(MATLAB_MEX_FILE) || defined(RSIM_WITH_SL_SOLVER)) && !defined(__linux__) 



304	genSimOutput(h, pln0, pln1, pln2);



305#else



306	readV4L2Frame(h, pln0, pln1, pln2);



307	if (h->pixelOrder == PIXEL_ORDER_PLANAR) {



308		/*interleaved2planar(pln0, pln0);*/



309	}



310#endif



311	h->frmCount++;



312}

314/* Clean-up routine that gets executed */



315/* when MW_ERROR_EXIT? macro is called */



316void videoCaptureCleanup(void) {



317	uint32_T i;



318



319	for (i = 0; i < MW_NUM_MAX_VIDEO_DEVICES; i++) {



320#ifdef _MW_V4L2_DEBUG_LVL2_



321        printf("Cleaning up resources for %s. PID = %d\n", hVideoDev[i].devName, getpid());



322#endif



323		terminateDevice(&hVideoDev[i]);



324	}



325}
