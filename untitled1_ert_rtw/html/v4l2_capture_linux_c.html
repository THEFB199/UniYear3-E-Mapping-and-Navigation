82/* Wrapper for IOCTL calls */



83static int xioctl(int fd, int request, void *arg)



84{



85    int res;



86    int tries = 3;



87



88    do {



89        res = ioctl(fd, request, arg);



90    } while ((res == -1) && (tries-- > 0) && ((errno == EINTR) || (errno == EAGAIN) || (errno == ETIMEDOUT)));



91



92    return res;



93}

247/* Open V4L2 device file */



248static void openV4L2Device(MW_videoInfo_t *h)



249{



250    struct stat st;



251



252    if (stat((const char *)h->devName, &st) == -1) {



253        MW_ERROR_EXIT3("Error: There is no video device called '%s'. System returned (%d): %s. Make sure that you specified the correct video device name.\n", 



254            h->devName, errno, strerror(errno));



255    }



256    if (!S_ISCHR(st.st_mode)) {



257        MW_ERROR_EXIT1("Error: %s is not a video capture device. Make sure that you specified the correct video device name.\n", h->devName);



258    }



259



260    /* Open device */



261    h->fd = open((const char *)h->devName, O_RDWR, 0);



262    if (h->fd == -1) {



263        MW_ERROR_EXIT3("Error: Cannot open video device file %s: System returned (%d): %s. "



264               "Make sure that you have read / write permissions to access the video device file and that no other application is using the video device.\n",



265                h->devName, errno, strerror(errno));



266    }



267}

269static void setFrameRate(MW_videoInfo_t *h)



270{ 



271    struct v4l2_streamparm streamparm;



272    int ret;



273



274    streamparm.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;



275    ret = xioctl(h->fd, VIDIOC_G_PARM, &streamparm);



276    if (ret < 0) {



277	return;



278    }



279



280    if (!(streamparm.parm.capture.capability & V4L2_CAP_TIMEPERFRAME)) {



281        return;



282    }



283    streamparm.parm.capture.timeperframe.numerator   = 1;



284    streamparm.parm.capture.timeperframe.denominator = h->frameRate; 



285    ret = xioctl(h->fd, VIDIOC_S_PARM, &streamparm);



286    if (ret < 0) {



287        return;



288    } 



289    



290#ifdef _MW_V4L2_DEBUG_



291    /* Get frame rate to confirm */



292    ret = xioctl(h->fd, VIDIOC_G_PARM, &streamparm);



293    if (ret < 0) {



294        return;



295    } 



296    printf("Frame rate: %d/%d\n", 



297            streamparm.parm.capture.timeperframe.denominator, 



298            streamparm.parm.capture.timeperframe.numerator);



299#endif  



300}

302/* Init device for video capture */



303static void initV4L2Device(MW_videoInfo_t *h)



304{



305    struct v4l2_capability cap;



306    struct v4l2_format fmt;



307#ifdef _V4L2_ENABLE_VIDEO_CROP_



308    struct v4l2_cropcap cropcap;



309    struct v4l2_crop crop;



310#endif



311



312    /* Query device capabilities */



313    memset(&cap, 0, sizeof(cap));



314    if (xioctl(h->fd, VIDIOC_QUERYCAP, &cap) == -1) {



315        if (EINVAL == errno) {



316            MW_ERROR_EXIT1("Error: %s is not a V4L2 video capture device.\n", h->devName);



317        }



318        else {



319            MW_ERROR_EXIT3("Error: Cannot query '%s': System returned (%d): %s.\n",



320                h->devName, errno, strerror(errno));



321        }



322    }



323



324    /* Check if device supports video capture */



325    if (!(cap.capabilities & V4L2_CAP_VIDEO_CAPTURE)) {



326#ifdef _MW_V4L2_DEBUG_



327        listDeviceCapabilities(h, &cap);



328#endif



329        MW_ERROR_EXIT1("Error: %s does not support video capture.\n", 



330                 h->devName);



331    }



332



333    /* Check that device has streaming or read/write capability */



334    if (cap.capabilities & V4L2_CAP_STREAMING) {



335        h->ioMethod = IO_METHOD_MMAP;



336    }



337    else {



338        MW_ERROR_EXIT1("Error: %s does not support STREAMING mode for video capture.\n", h->devName);



339    }



340



341#ifdef _V4L2_ENABLE_VIDEO_CROP_



342    /* Get cropping parameters */



343    memset(&cropcap, 0, sizeof(cropcap));



344    cropcap.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;



345    if (xioctl(h->fd, VIDIOC_CROPCAP, &cropcap) == -1) {



346        MW_ERROR_EXIT3("Error: %s does not support crop capability: System returned (%d): %s.\n", 



347                h->devName, errno, strerror(errno));



348    }



349#ifdef _MW_V4L2_DEBUG_



350    printCropCap(&cropcap);



351#endif



352



353    /* Set region of interest */



354    memset(&crop, 0, sizeof(crop));



355    crop.type     = V4L2_BUF_TYPE_VIDEO_CAPTURE;



356    crop.c.top    = h->roi.top; 



357	crop.c.left   = h->roi.left;



358    crop.c.width  = h->roi.width;



359	crop.c.height = h->roi.height;



360    if (xioctl(h->fd, VIDIOC_S_CROP, &crop) == 0) {



361        /* Read back */



362        if (xioctl(h->fd, VIDIOC_G_CROP, &crop) == -1) {



363            MW_ERROR_EXIT3("Error: %s does not support GET crop capability IOCTL. System returned (%d): %s. .\n", 



364                     h->devName, errno, strerror(errno));



365        }



366



367        /* Ensure that the requested cropping area is set  */



368        /* fixme */



369#ifdef _MW_V4L2_DEBUG_



370        printf("Region of interest set to: [%d, %d, %d, %d].\n",



371                  crop.c.top, crop.c.left, crop.c.width, crop.c.height);



372#endif



373    }



374    else {



375        /*EINVAL: Device does not support cropping */



376#ifdef _MW_V4L2_DEBUG_



377        printf("Error: %s does not support SET crop capability IOCTL. System returned (%d): %s.\n", 



378                  h->devName, errno, strerror(errno));



379#endif



380    }



381#endif /* _V4L2_ENABLE_VIDEO_CROP_ */



382



383    /* Set data format and images size */



384    memset(&fmt, 0, sizeof(fmt));



385    fmt.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;



386    if (xioctl(h->fd, VIDIOC_G_FMT, &fmt) == -1) {



387        MW_ERROR_EXIT3("Error: Cannot get current video format of %s: System returned (%d): %s.\n", 



388                  h->devName, errno, strerror(errno));



389    }



390#ifdef _MW_V4L2_DEBUG_



391    printPixFormat(&fmt.fmt.pix);



392#endif /* _V4L2_ENABLE_VIDEO_CROP_ */



393   



394    /* Try selected format */



395    if ((fmt.fmt.pix.width != h->imFormat.width) || 



396            (fmt.fmt.pix.height != h->imFormat.height) || 



397            (fmt.fmt.pix.pixelformat != V4L2_PIX_FMT_YUYV))



398    {



399        fmt.fmt.pix.width       = h->imFormat.width;



400        fmt.fmt.pix.height      = h->imFormat.height;



401        fmt.fmt.pix.pixelformat = V4L2_PIX_FMT_YUYV; 



402        if (xioctl(h->fd, VIDIOC_S_FMT, &fmt) == -1) {



403            switch (errno) {



404                case EBUSY:



405                    MW_ERROR_EXIT3("Error: %s is not available for I/O. System returned (%d): %s."



406                            "Make sure that device is not used by another application.\n",



407                            h->devName, errno, strerror(errno));



408                    break;



409                case EINVAL:



410                {   struct v4l2_fmtdesc fmtDesc;



411                    char *errMsgPtr = &strBuf[0];



412                    size_t size = ERR_MSG_BUF_SIZE;



413                    size_t ret;



414                    



415                    /* Print out format description */



416                    fmtDesc.type  = V4L2_BUF_TYPE_VIDEO_CAPTURE;



417                    fmtDesc.index = 0;



418                    while ((xioctl(h->fd, VIDIOC_ENUM_FMT, &fmtDesc) != -1) && (size > 0)) {



419                        ret = printFormatDescription(&fmtDesc, errMsgPtr, size);



420                        errMsgPtr += ret;  /* Tag on \0 at the end */



421                        size      -= ret;



422                        fmtDesc.index += 1;



423                    }



424                    MW_ERROR_EXIT2("Error: %s pixel format is not supported by the device. "



425                            "Pixel formats supported by device: \n%s\n",



426                            pixFmt2Name(fmt.fmt.pix.pixelformat), strBuf);



427                }



428                break;



429                default:



430                    MW_ERROR_EXIT3("Error: VIDIOC_S_FMT call failed for device %s: %d, %s.",



431                            h->devName, errno, strerror(errno));



432                    break;



433            }



434        }



435    }



436



437    /* Note VIDIOC_S_FMT may change width and height. */



438    if ((h->imFormat.width != fmt.fmt.pix.width) || 



439        (h->imFormat.height != fmt.fmt.pix.height)) {



440        MW_ERROR_EXIT4("Error: Image size of [%d, %d] is not supported. "



441               "The closest image size supported by the device is [%d, %d].",



442                h->imFormat.width, h->imFormat.height,



443                fmt.fmt.pix.width, fmt.fmt.pix.height);



444    }



445



446    /* Set frame rate */



447    setFrameRate(h);



448    



449    /* Initialize device for I/O */



450    initMmap(h);



451



452    /* Start data capture */



453    startCapture(h);



454}

456/* Stop video capture */



457static void stopCapture(MW_videoInfo_t *h)



458{



459    enum v4l2_buf_type type;



460



461    switch (h->ioMethod) {



462        case IO_METHOD_READ:



463            /* Nothing to do. */



464            break;



465        case IO_METHOD_MMAP:



466        case IO_METHOD_USERPTR:



467            type = V4L2_BUF_TYPE_VIDEO_CAPTURE;



468            if (xioctl(h->fd, VIDIOC_STREAMOFF, &type) == -1) {



469                MW_ERROR_EXIT("Error: Cannot stop video capture.");



470            }



471            h->v4l2CaptureStarted = 0;



472            break;



473    }



474}

476/* Start capturing video frames */



477static void startCapture(MW_videoInfo_t *h)



478{



479    unsigned int i;



480    enum v4l2_buf_type type;



481



482    switch (h->ioMethod) {



483        case IO_METHOD_USERPTR:



484        case IO_METHOD_READ:



485            /* Nothing to do. */



486            break;



487        case IO_METHOD_MMAP:



488            for (i = 0; i < h->v4l2BufCount; ++i) {



489                struct v4l2_buffer buf;



490



491                memset(&buf, 0, sizeof(buf));



492                buf.type   = V4L2_BUF_TYPE_VIDEO_CAPTURE;



493                buf.memory = V4L2_MEMORY_MMAP;



494                buf.index  = i;



495                if (xioctl(h->fd, VIDIOC_QBUF, &buf) == -1) {



496                    MW_ERROR_EXIT2("Error queuing video buffer. System returned (%d): %s.", errno, strerror(errno));



497                }



498            }



499                 



500            /* Start video stream */



501            type = V4L2_BUF_TYPE_VIDEO_CAPTURE;



502            if (xioctl(h->fd, VIDIOC_STREAMON, &type) == -1) {



503                MW_ERROR_EXIT2("Error starting capture. System returned (%d): %s.", errno, strerror(errno));



504            }



505            h->v4l2CaptureStarted = 1;



506            break;



507     }



508}

511/* Initialize MMAP operation */



512static void initMmap(MW_videoInfo_t *h)



513{



514    struct v4l2_requestbuffers req;



515    unsigned int i;



516



517    /* Request MMAP buffers */



518    memset(&req, 0, sizeof(req));



519    req.count  = MW_NUM_V4L2_BUFFERS;



520    req.type   = V4L2_BUF_TYPE_VIDEO_CAPTURE;



521    req.memory = V4L2_MEMORY_MMAP;



522    if (xioctl(h->fd, VIDIOC_REQBUFS, &req) == -1) {



523        if (errno == EINVAL) {



524            MW_ERROR_EXIT1("Error: %s does not support memory mapping.\n", h->devName);



525        } 



526        else {



527            MW_ERROR_EXIT("Error requesting MMAP buffers.");



528        }



529    }



530#ifdef _MW_V4L2_DEBUG_



531    printf("Number of MMAP buffers = %d\n", req.count);



532#endif



533



534    /* Check number of buffers */



535    if (req.count < 2) {



536        MW_ERROR_EXIT1("Error: Insufficient buffer memory on %s\n", h->devName);



537    }



538    h->v4l2BufCount = req.count; /* Record buffer count for feature reference */



539



540    /* Queue all buffer to device */



541    for (i = 0; i < h->v4l2BufCount; i++) {



542        struct v4l2_buffer buf;



543



544        memset(&buf, 0, sizeof(buf));



545        buf.type   = V4L2_BUF_TYPE_VIDEO_CAPTURE;



546        buf.memory = V4L2_MEMORY_MMAP;



547        buf.index  = i;



548        if (xioctl(h->fd, VIDIOC_QUERYBUF, &buf) == -1) {



549            MW_ERROR_EXIT("Error querying video buffer.");



550        }



551        h->frm[i].start  = mmap(NULL, buf.length, PROT_READ | PROT_WRITE,



552                                MAP_SHARED, h->fd, buf.m.offset); 



553        if (h->frm[i].start == MAP_FAILED) {



554            MW_ERROR_EXIT("Error: mmap operation failed.");



555        }



556        h->frm[i].length = buf.length;



557    }



558}

560/* Convert from YUYV (interleaved) format to YUV planar */



561static void yuyv2yuv(uint8_T *yuyv, uint8_T *y, uint8_T *u, uint8_T *v, uint32_T width, uint32_T height)



562{



563    uint32_T i;



564



565    for (i = 0; i < (height * (width >> 1)); i++) {



566        *y++ = *yuyv++;



567        *u++ = *yuyv++;



568        *y++ = *yuyv++;



569        *v++ = *yuyv++;



570    }



571}

573#define RCr    358  /* (1.402 * 255) */



574#define GCb    88   /* (0.34414 * 255) */



575#define GCr    182  /* (0.71414 * 255) */



576#define BCb    452  /* (1.772 * 255) */



577static uint8_T clamp(int16_T x)  



578{



579    /*(x > 255) ? 255 : ((x < 0) ? 0 : x);*/



580    if (x > 255) {



581        x = 255;



582    } else if (x < 0) {



583        x = 0;



584    }



585    



586    return (x);



587}



588/* Convert from YUYV (interleaved) format to YUV planar */



589static void yuyv2rgb(uint8_T *yuyv, uint8_T *r, uint8_T *g, uint8_T *b, uint32_T width, uint32_T height)



590{



591    uint32_T i;



592    int16_T Y0, Y1, Cb, Cr;



593



594    for (i = 0; i < (height * (width >> 1)); i++) {



595        /* Y0 U0 Y1 V0 Y2 U1 Y3 V1 */



596        /* MSDN



597         C = Y - 16



598         D = U - 128



599         E = V - 128



600         R = clip(( 298 * C           + 409 * E + 128) >> 8)



601         G = clip(( 298 * C - 100 * D - 208 * E + 128) >> 8)



602         B = clip(( 298 * C + 516 * D           + 128) >> 8)



603         



604         Formula used here is:



605         R = Y + 1.402 (Cr-128)



606         G = Y - 0.34414 (Cb-128) - 0.71414 (Cr-128)



607         B = Y + 1.772 (Cb-128)



608        */



609        Y0 = *yuyv++;



610        Cb = *yuyv++ - 128;



611        Y1 = *yuyv++;



612        Cr = *yuyv++ - 128;



613        



614        *r++ = clamp(Y0 + ((RCr * Cr) >> 8));



615        *g++ = clamp(Y0 - ((GCb * Cb + GCr * Cr) >> 8));



616        *b++ = clamp(Y0 + ((BCb * Cb) >> 8));



617        *r++ = clamp(Y1 + ((RCr * Cr) >> 8));



618        *g++ = clamp(Y1 - ((GCb * Cb + GCr * Cr) >> 8));



619        *b++ = clamp(Y1 + ((BCb * Cb) >> 8));



620    }



621}

623/* Convert from RGB24 (interleaved) format to RGB planar */



624static void rgb242rgb(uint8_T *rgb24, uint8_T *r, uint8_T *g, uint8_T *b, uint32_T width, uint32_T height)



625{



626    uint32_T i;



627



628    for (i = 0; i < (height * width); i++) {



629        *r++ = *rgb24++;



630        *g++ = *rgb24++;



631        *b++ = *rgb24++;



632    }



633}

635/* Reads a single video frame from a V4L2 device */



636static void readV4L2Frame(MW_videoInfo_t *h, uint8_T *pln0, uint8_T *pln1, uint8_T *pln2)



637{



638    struct v4l2_buffer buf;



639



640    switch (h->ioMethod) {



641        case IO_METHOD_USERPTR:



642        case IO_METHOD_READ:



643            /* Nothing to do. */



644            break;



645        case IO_METHOD_MMAP:



646            /* Request a new buffer */



647            memset(&buf, 0, sizeof(buf));



648            buf.type   = V4L2_BUF_TYPE_VIDEO_CAPTURE;



649            buf.memory = V4L2_MEMORY_MMAP;



650            if (xioctl(h->fd, VIDIOC_DQBUF, &buf) == -1) {



651                switch (errno) {



652                    case EAGAIN:



653                    case EIO:



654                    default:



655                        printf("Error: Failure reading image buffer. System returned (%d): %s.\n", 



656                                errno, strerror(errno));



657                }



658             }



659             if (buf.index >= h->v4l2BufCount) {



660                  MW_ERROR_EXIT("Error: Frame buffer index out of range.");



661             }



662



663             /* Copy acquired frame to output signal line */



664             if (h->pixelOrder == PIXEL_ORDER_PLANAR) {



665                 switch(h->imFormat.pixelFormat) {



666                     case MW_YCBCR422:



667                         yuyv2yuv((uint8_T *)h->frm[buf.index].start, pln0, pln1, pln2, 



668                                   h->imFormat.width, h->imFormat.height);



669                         break;



670                     case MW_RGB:



671                         yuyv2rgb((uint8_T *)h->frm[buf.index].start, pln0, pln1, pln2, 



672                                    h->imFormat.width, h->imFormat.height);



673                         break;



674                 }



675             }



676             else {



677                 /* Note pln1 and pln2 are not updated */



678                 memcpy(pln0, h->frm[buf.index].start, (h->imFormat.width) * (h->imFormat.height));



679             }



680



681             /* Return buffer back to device */



682             if (xioctl(h->fd, VIDIOC_QBUF, &buf) == -1) {



683                 printf("Error: Failure returning video buffer to device %s. System returned (%d): %s.\n", 



684                         h->devName, errno, strerror(errno));



685             }



686             break;



687    }



688}

690/* Close V4L2 device */



691static void closeV4L2Device(MW_videoInfo_t *h)



692{



693    unsigned int i;



694    struct v4l2_requestbuffers req;



695    



696    if ( (h != NULL) && (h->fd > 0)) {



697        /* Stop streaming */



698        if (h->v4l2CaptureStarted == 1) {



699            /* All buffers are unqueued when capture is stopped */



700            /* http://linuxtv.org/downloads/v4l-dvb-apis/vidioc-streamon.html */



701            stopCapture(h);



702        }



703        



704        /* Unmap / free buffers allocated for capture */



705        switch (h->ioMethod) {



706            case IO_METHOD_USERPTR:



707            case IO_METHOD_READ:



708                /* Nothing to do. */



709                break;



710            case IO_METHOD_MMAP:



711                /* Unmap all buffers */



712                for (i = 0; i < h->v4l2BufCount; ++i) {



713                    if (munmap(h->frm[i].start, h->frm[i].length) == -1) {



714                        printf("Warning: munmap call failed. System returned (%d): %s.",



715                                errno, strerror(errno));



716                    }



717                }



718



719                /* Delete buffers */



720                memset(&req, 0, sizeof(req));



721                req.count  = 0;



722                req.type   = V4L2_BUF_TYPE_VIDEO_CAPTURE;



723                req.memory = V4L2_MEMORY_MMAP;



724                if(xioctl(h->fd, VIDIOC_REQBUFS, &req) < 0) {



725                    printf("Failed to delete video buffers: System returned (%d): %s.\n",



726                            errno, strerror(errno));



727                }



728                break;



729        }



730        



731        /* Close device file */



732        if (close(h->fd) == -1) {



733            printf("Warning: cannot close device file %s. System returned (%d): %s.",



734                    h->devName, errno, strerror(errno));



735        }



736        h->fd = -1;



737        sleep(1);



738    }



739}
