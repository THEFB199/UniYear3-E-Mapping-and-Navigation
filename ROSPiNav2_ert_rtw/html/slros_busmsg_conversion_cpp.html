65void convertFromBus(sensor_msgs::Joy* msgPtr, SL_Bus_ROSPiNav2_sensor_msgs_Joy const* busPtr)



66{



67  const std::string rosMessageType("sensor_msgs/Joy");

69  convertFromBusVariablePrimitiveArray(msgPtr->axes, busPtr->Axes, busPtr->Axes_SL_Info);



70  convertFromBusVariablePrimitiveArray(msgPtr->buttons, busPtr->Buttons, busPtr->Buttons_SL_Info);



71  convertFromBus(&msgPtr->header, &busPtr->Header);



72}

77
