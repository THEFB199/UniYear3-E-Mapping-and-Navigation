85        /* Open SPI channel */



86        static int SPI_open(const unsigned int slaveSelectPin)



87        {



88            int fd;



89            char buf[MAX_BUF_SIZE];



90



91            snprintf(buf, sizeof(buf), SPI_DEV_FILE "%d", slaveSelectPin);



92            fd = open(buf, O_RDWR);



93            if (fd < 0) {



94                fprintf(stderr,"%s\n", buf);



95                perror("SPI_open/open");



96            }



97            return fd;



98        }

100        /* Close SPI channel */



101        static void SPI_close(int fd)



102        {



103            int ret;



104            printf("SPI_close fd = %d \n",fd);



105            fflush(stdout);



106            ret = close(fd);



107            if (ret < 0) {



108                /* EBADF, EINTR, EIO: In all cases, descriptor is torn down*/



109                perror("SPI_close/close");



110            }

114        /* SVD APIs */



115        MW_Handle_Type MW_SPI_Open(



116                uint32_T SPIModule, 



117                uint32_T MosiPin,        /* Not used*/



118                uint32_T MisoPin,         /* Not used */



119                uint32_T ClockPin,        /* Not used */



120                uint32_T SlaveSelectPin, 



121                uint8_T ActiveLowSSPin, 



122                uint8_T IsSPIDeviceSlave)



123        {



124            int ret;



125            SPI_dev_t *spi;



126            MW_Handle_Type SPIModuleHandle = (MW_Handle_Type)NULL;



127



128           /* Check parameters */



129            if (IsSPIDeviceSlave) {



130                fprintf(stderr,"Only SPI master mode is supported.\n");



131                exit(-1);



132            }



133



134            if (SPIModule != 0) {



135                fprintf(stderr,"MW_SPI_Open:SPI module must be set to 0.\n");



136                exit(-1);



137            }



138



139            if ((SlaveSelectPin != 0) && (SlaveSelectPin != 1)) {



140                fprintf(stderr,"SlaveSelectPin must be 0 or 1.\n");



141                exit(-1);



142            }



143			/* Current slave select */



144			currSpiDev = SlaveSelectPin;



145			



146            /* Get device handle */



147            spi = &spiDev[SlaveSelectPin];



148            



149            if (spi->fd < 0) {



150                spi->fd = SPI_open(SlaveSelectPin);



151                if (spi->fd < 0) {



152                    fprintf(stderr,"Error opening SPI bus.\n");



153                    exit(-1);



154                }



155



156                /* Return handle */



157                SPIModuleHandle = (MW_Handle_Type)spi;



158            }



159            else if (spi->fd > 0)



160            {



161                /* Return handle */



162                SPIModuleHandle = (MW_Handle_Type)spi;



163            }



164



165            return SPIModuleHandle;



166        }

168        MW_SPI_Status_Type MW_SPI_SetSlaveSelect(



169                MW_Handle_Type SPIModuleHandle, 



170                uint32_T SlaveSelectPin, 



171                uint8_T ActiveLowSSPin)



172        {



173            MW_SPI_Status_Type status;



174



175            if (SPIModuleHandle == (MW_Handle_Type)NULL) {



176                status = MW_SPI_BUS_NOT_AVAILABLE;



177            }



178            else if (((SPI_dev_t *)SPIModuleHandle)->SlaveSelectPin != SlaveSelectPin) {



179                status = MW_SPI_BUS_NOT_AVAILABLE;



180            }



181            else if (ActiveLowSSPin != 1) {



182                status = MW_SPI_BUS_NOT_AVAILABLE;



183            }



184            else



185            {



186                currSpiDev = SlaveSelectPin;



187                status = MW_SPI_SUCCESS;



188            }



189



190            return status;



191        }

194        MW_SPI_Status_Type MW_SPI_SetFormat(



195                MW_Handle_Type SPIModuleHandle, 



196                uint8_T TargetPrecision,   /* Only 8-bit supported*/



197                MW_SPI_Mode_type SPIMode,  /* 0 - 3*/



198                MW_SPI_FirstBitTransfer_Type TargetFirstBitToTransfer)



199        {



200            SPI_dev_t *spi;



201            int ret;



202



203            spi = (SPI_dev_t *)SPIModuleHandle;



204            



205            if (SPIModuleHandle == (MW_Handle_Type)NULL)



206            {



207                return MW_SPI_BUS_NOT_AVAILABLE;



208            }



209



210            /* We are assuming here that the currSpiDev has been set via a call to



211             MW_SPI_SetSlaveSelect */



212            /* spi = &spiDev[currSpiDev]; */



213



214            /* Set mode */



215            ret = ioctl(spi->fd, SPI_IOC_WR_MODE, &SPIMode);



216            if (ret < 0){



217                perror("SPI_init/SPI_IOC_WR_MODE");



218                return MW_SPI_BUS_ERROR;



219            }



220            ret = ioctl(spi->fd, SPI_IOC_RD_MODE, &SPIMode);



221            if (ret < 0) {



222                perror("SPI_init/SPI_IOC_RD_MODE");



223                return MW_SPI_BUS_ERROR;



224            }



225



226            /* Set bits per word*/



227            ret = ioctl(spi->fd, SPI_IOC_WR_BITS_PER_WORD, &TargetPrecision);



228            if (ret < 0) {



229                perror("SPI_init/SPI_IOC_WR_BITS_PER_WORD");



230                return MW_SPI_BUS_ERROR;



231            }



232            ret = ioctl(spi->fd, SPI_IOC_RD_BITS_PER_WORD, &TargetPrecision);



233            if (ret < 0) {



234                perror("SPI_init/SPI_IOC_RD_BITS_PER_WORD");



235                return MW_SPI_BUS_ERROR;



236            }



237            /* Ignore TargetFirstBitToTransfer as Raspberry Pi always uses MSB first*/



238            (void)TargetFirstBitToTransfer;



239



240            return MW_SPI_SUCCESS;



241        }

243        MW_SPI_Status_Type MW_SPI_SetBusSpeed(



244                MW_Handle_Type SPIModuleHandle, 



245                uint32_T BusSpeedInHz)



246        {



247            SPI_dev_t *spi;



248            int ret;



249



250            spi = (SPI_dev_t *)SPIModuleHandle;



251



252            if (SPIModuleHandle == (MW_Handle_Type)NULL) {



253                return MW_SPI_BUS_NOT_AVAILABLE;



254            }



255



256            /* We are assuming here that the currSpiDev has been set via a call to



257             MW_SPI_SetSlaveSelect or MW_SPI_Open */



258			if (0 == spi->SlaveSelectPin)



259			{



260				#ifdef MW_SPI_SPI0CE0BUSSPEED



261                BusSpeedInHz = BusSpeed[MW_SPI_SPI0CE0BUSSPEED]*1000;



262                #endif



263			}



264			else if (1 == spi->SlaveSelectPin)



265            {



266                #ifdef MW_SPI_SPI0CE1BUSSPEED



267                BusSpeedInHz = BusSpeed[MW_SPI_SPI0CE1BUSSPEED]*1000;



268                #endif



269            }



270			else



271            {



272                /* Consider BusSpeedInHz from function argument */



273            }



274            



275            ret = ioctl(spi->fd, SPI_IOC_WR_MAX_SPEED_HZ, &BusSpeedInHz);



276            if (ret < 0) {



277                perror("SPI_init/SPI_IOC_WR_MAX_SPEED_HZ");



278                return MW_SPI_BUS_NOT_AVAILABLE;



279            }



280            ret = ioctl(spi->fd, SPI_IOC_RD_MAX_SPEED_HZ, &BusSpeedInHz);



281            if (ret < 0) {



282                perror("SPI_init/SPI_IOC_RD_MAX_SPEED_HZ");



283                return MW_SPI_BUS_NOT_AVAILABLE;



284            }



285



286            return MW_SPI_SUCCESS;



287        }

289        MW_SPI_Status_Type MW_SPI_MasterWriteRead_8bits(



290                MW_Handle_Type SPIModuleHandle, 



291                const uint8_T * WriteDataPtr, 



292                uint8_T * ReadDataPtr, 



293                uint32_T DataLength)



294        {



295            SPI_dev_t *spi;



296            int ret;



297            struct spi_ioc_transfer tr;



298



299            spi = (SPI_dev_t *)SPIModuleHandle;



300



301            if (SPIModuleHandle == (MW_Handle_Type)NULL) {



302                return MW_SPI_BUS_NOT_AVAILABLE;



303            }



304



305            if (DataLength > 4096) {



306                /* SPI data transfer size is limited to page size which is usually 4096 byte*/



307                return MW_SPI_BUS_NOT_AVAILABLE;



308            }



309



310            /* We are assuming here that the currSpiDev has been set via a call to



311             MW_SPI_SetSlaveSelect*/



312            /* spi = &spiDev[currSpiDev]; */



313



314            /* Setup SPI transfer structure*/



315            memset(&tr, 0, sizeof(struct spi_ioc_transfer));



316            tr.tx_buf = (unsigned long)WriteDataPtr;



317            tr.rx_buf = (unsigned long)ReadDataPtr;



318            tr.len    = DataLength;



319



320            /* Execute IOCTL call to perform a full-duplex transfer*/



321            ret = ioctl(spi->fd, SPI_IOC_MESSAGE(1), &tr);



322            if (ret != DataLength) {



323                perror("EXT_SPI_writeRead/ioctl");



324                return MW_SPI_BUS_ERROR;



325            }



326



327            return MW_SPI_SUCCESS;



328        }
